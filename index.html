// 言語切り替え function setLanguage(lang) { currentLanguage = lang; localStorage.setItem('language', lang); document.documentElement.setAttribute('data-language', lang); // すべての要素を走査して翻訳 const elements = document.querySelectorAll('[data-ja][data-en][data-ru]'); elements.forEach(el => { // 要素の最初の子ノード（通常はテキストノード）のみを置き換える let translatedText = ''; if (lang === 'ja' && el.dataset.ja) { translatedText = el.dataset.ja; } else if (lang === 'en' && el.dataset.en) { translatedText = el.dataset.en; } else if (lang === 'ru' && el.dataset.ru) { translatedText = el.dataset.ru; } // 最初のテキストノードだけを置き換える if (translatedText) { // 既存の最初のテキストノードを置き換え const firstChild = el.firstChild; if (firstChild && firstChild.nodeType === Node.TEXT_NODE) { firstChild.textContent = translatedText; } else { // テキストノードがない場合は作成 el.insertBefore(document.createTextNode(translatedText), firstChild); } } }); // ボタンテキストを更新 const langButton = document.querySelector('.lang-toggle'); const flags = { 'ja': '🇯🇵 日本語', 'en': '🇺🇸 English', 'ru': '🇷🇺 Русский' }; langButton.textContent = '🌐 ' + (flags[lang] || flags['ja']); // アクティブなボタンを更新 document.querySelectorAll('.lang-option').forEach(btn => { btn.classList.remove('active'); }); document.querySelector(`.lang-option[onclick="setLanguage('${lang}')"]`)?.classList.add('active'); // ドロップダウンを閉じる document.getElementById('langDropdown').classList.remove('active'); }
